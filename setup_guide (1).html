📊 Useful Admin Queries

Supabase SQL Editor mein run karo:

All Enrolled Users

SELECT p.name, p.created_at, COUNT(DISTINCT dc.day_id) as days_completed FROM profiles p LEFT JOIN day_completions dc ON p.id = dc.user_id GROUP BY p.id, p.name, p.created_at ORDER BY p.created_at DESC;

Leaderboard — Top Learners

SELECT p.name, COUNT(DISTINCT dc.day_id) as days_done, COALESCE(c.score, 0) as final_score FROM profiles p LEFT JOIN day_completions dc ON p.id = dc.user_id LEFT JOIN certificates c ON p.id = c.user_id GROUP BY p.id, p.name, c.score ORDER BY days_done DESC, final_score DESC;

Total Stats

SELECT (SELECT COUNT(*) FROM profiles) as total_users, (SELECT COUNT(DISTINCT user_id) FROM certificates) as certified, (SELECT COUNT(*) FROM progress WHERE completed = true) as total_tabs_done;
